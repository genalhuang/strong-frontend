### 链表
#### 特征： 
1. 链表是以节点的方式存储
2. 每个节点包含data域，next域：指向下一节点
3. 节点不一定连续存储


#### 单向链表
##### 单链表面试题
1. 查找单链表倒数第k个节点
   ```java
   思路
   1. 编写一个方法，接收head节点，同时接收index
   2. index代表倒数第k个
   3. 先把链表从头遍历1，得到链表长度
   4. 得到size后从链表的第一个开始遍历，遍历size-index个
   
   public static findlastindexnode(heronode head, int index) {
     if(head.next == null) {
       return null;
     }
     int size = getLength(head);
     if(index <= 0 || index > size) {
       return null;
     }
     heronode cur = head.next;
     for(int i = 0;i < size - index; i++) {
       cur = cur.next;
     }
     return cur;
   }
   ```
2.  单链表的反转
    ``` java
    思路
    1. 先定义一个节点reverseHead = new HeroNode();
    2. 从头到尾遍历原来的链表，每遍历一个节点，就将其取出，并放在新的链表reverseHead的最前端
    3. 原来的链表的head.next = reverseHead.next
   
    public static void reversetList(HeroNode head) {
      if(head.next == null || head.next.next == null) {
        return ;
      }
      // 定义一个辅助指针，帮助我们遍历原来的链表
      HeroNode cur = head.next;
      HeroNode next = null; //指向当前节点的下一个节点；
      HeroNode reverseHead = new HeroNode(0,"","");
      // 遍历原来的链表
      while(cur != null) {
        next = cur.next;
        cur.next = reverseHead.next;
        reverseHead.next = cur;
        cur = next
      }
      head.next = reverseHead.next;
    }
    ```
3.  从尾到头打印单链表
    ```java
    思路
    1. 方式1：先将单链表进行反转，再遍历打印（不建议）
    2. 方式2：利用栈，将各个节点压入栈中，利用栈的先进后出实现
    ```

```java
// 定义MySingleLinkedList 管理我们的英雄
class MySingleLinkedList {
	// 初始化头结点
	private HeroNode head = new HeroNode(0, "", "");

	// 添加链表
	public void add(HeroNode heroNode) {
		HeroNode temp = head;
		while (true) {
			if (temp.next == null) {
				break;
			}
			temp = temp.next;
		}
		temp.next = heroNode;
	}

	// 显示链表
	public void list() {
		if (head.next == null) {
			System.out.print("链表空");
			return;
		}
		HeroNode temp = head.next;
		while (true) {
			if (temp == null) {
				break;
			}
			System.out.print(temp);
			temp = temp.next;
		}
	}

	// 添加英雄是，根据排名插入英雄
	public void addByOrder(HeroNode heroNode) {
		HeroNode temp = head;
		boolean flag = false;
		while (true) {
			if (temp.next == null) {
				break;
			} else if (temp.next.no > heroNode.no) {
				break;
			} else if (temp.next.no == heroNode.no) {
				flag = true;
				break;
			}
			temp = temp.next;
		}
		if (flag) {
			System.out.print("编号被占用\n");
		} else {
			heroNode.next = temp.next;
			temp.next = heroNode;
		}

	}

	// 修改英雄信息
	public void update(HeroNode newHeroNode) {
		if (head.next == null) {
			System.out.println("链表空");
			return;
		}
		HeroNode temp = head.next;
		boolean flag = false;
		while (true) {
			if (temp == null) {
				break;
			}
			if (temp.no == newHeroNode.no) {
				flag = true;
				break;
			}
			temp = temp.next;
		}
		if (flag) {
			temp.name = newHeroNode.name;
			temp.nickname = newHeroNode.nickname;
		} else {
			System.out.printf("没找到编号%d的英雄节点", newHeroNode.no);
		}
	}

	// 删除节点
	public void delete(int no) {
		HeroNode temp = head;
		boolean flag = false;
		while (true) {
			if (temp.next == null) {
				break;
			}
			if (temp.next.no == no) {
				flag = true;
				break;
			}
			temp = temp.next;
		}
		if (flag) {
			temp.next = temp.next.next;
		} else {
			System.out.println("要删除的节点不存在");
		}
	}
}

// 定义英雄节点
class HeroNode {
	public int no;
	public String name;
	public String nickname;
	public HeroNode next;

	public HeroNode(int no, String name, String nname) {
		this.no = no;
		this.name = name;
		this.nickname = nname;
	}

	@Override
	public String toString() {
		return "HeroNode:" + "\t" + no + name + "\t" + nickname + "\n";
	}
}
```

#### 单向环形链表（约瑟夫问题）

```java
小孩成环状，第k个开始报数，第m个出去，m的下一个重新报数m次
n = 5 既5个人
k = 1 从第一个开始报数
m = 2 数2下
思路：
1. 先创建一个环形单链表
2. 创建辅助指针，指向环形链表最后一个节点
3. 将first和helper移动k-1下。
4. 小孩报数时，让first和helper同时移动m-1次
5. first指向的小孩节点出圈
   first = first.next
   helper.next = first
6. 如果first == helper说明出圈完毕

代码：
package linkedist;

public class Josepfu {

	public static void main(String[] args) {
		// 测试一把
		CircleSingleLinkedList a = new CircleSingleLinkedList();
		a.addBoy(5);
		a.showBoy();
		a.countBoy(1, 2, 5);

	}

}

// 创建一个环形的单向链表
class CircleSingleLinkedList {
	// 创建first节点
	private Boy first = null;

	// 新增
	public void addBoy(int nums) {
		if (nums < 1) {
			System.out.println("nums的值不正确");
			return;
		}
		Boy curBoy = null;
		for (int i = 1; i <= nums; i++) {
			Boy boy = new Boy(i);
			if (i == 1) {
				first = boy;
				curBoy = first;
				boy.setNext(first);
			} else {
				curBoy.setNext(boy);
				boy.setNext(first);
				curBoy = boy;
			}
		}
	}

	// 遍历
	public void showBoy() {
		if (first == null) {
			System.out.println("链表为空");
			return;
		}
		Boy curBoy = first;
		while (true) {
			System.out.printf("链表编号%d\n", curBoy.getNo());
			if (curBoy.getNext() == first) {
				break;
			}
			curBoy = curBoy.getNext();
		}
	}

	// 根据用户的输入，计算出小孩出圈的顺序
	/**
	 * @param startNo  从第几个小孩开始数
	 * @param countNum 数几下
	 * @param nums     表示最初有多少个小孩在圈中
	 */
	public void countBoy(int startNo, int countNum, int nums) {
		// 数据校验
		if (first == null || startNo < 1 || startNo > nums) {
			System.out.println("参数输入有误，请重新输入");
			return;
		}
		// 创建辅助指针
		Boy helper = first;
		while (true) {
			if (helper.getNext() == first) {
				break;
			}
			helper = helper.getNext();
		}
		// 移动到开始出圈的位置
		for(int j = 0; j < startNo - 1; j++) {
			first = first.getNext();
			helper = helper.getNext();
		}
		// 开始出圈
		while(true) {
			if(helper == first) {
				break;
			}
			// 让first helper同时移动countNum-1次
			for(int j = 0; j < countNum - 1; j++) {
				first = first.getNext();
				helper = helper.getNext();
			}
			// 此时first节点就是出圈小孩的节点
			System.out.printf("小孩%d出圈\n",first.getNo());
			// 将first指向的小孩节点出圈
			first = first.getNext();
			helper.setNext(first);
		}
		System.out.printf("最后一位小孩%d出圈\n",first.getNo());
	}
}

// 创建一个Boy类
class Boy {
	private int no;
	private Boy next;

	public Boy(int no) {
		this.no = no;
	}

	public int getNo() {
		return no;
	}

	public void setNo(int no) {
		this.no = no;
	}

	public Boy getNext() {
		return next;
	}

	public void setNext(Boy next) {
		this.next = next;
	}
}
```
