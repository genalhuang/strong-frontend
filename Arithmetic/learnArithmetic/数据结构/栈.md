### 栈

#### 特征：
1. 先入后出
   
#### 场景：
1. 子程序调用
2. 递归调用
3. 表达式转换与求值
4. 二叉树遍历
5. 图形的深度优先搜索法
   
#### 实现栈：
1. 使用数组
2. 定义top变量表示栈顶
3. 入栈操作
4. 出栈操作

```java
class ArrayStack {
	private int maxSize;
	private int[] stack;
	private int top = -1;

	public ArrayStack(int maxSize) {
		this.maxSize = maxSize;
		stack = new int[this.maxSize];
	}

	public boolean isFull() {
		return top == maxSize - 1;
	}

	public boolean isEmpty() {
		return top == -1;
	}

	// 入栈
	public void push(int value) {
		if (isFull()) {
			System.out.println("栈满");
			return;
		}
		top++;
		stack[top] = value;
	}

	// 出栈
	public int pop() {
		if (isEmpty()) {
			throw new RuntimeException("栈 空");
		}
		int value = stack[top];
		top--;
		return value;
	}

	// 遍历栈 从栈顶开始显示数据
	public void list() {
		if (isEmpty()) {
			System.out.println("栈空");
			return;
		}
		for (int i = top; i >= 0; i--) {
			System.out.printf("stack[%d]\n", stack[i]);
		}
	}
}
```


#### 练习
栈实现综合计算器
思路：
例如 7*2*2-5+1-5+3-4
1. 创建数栈和符号栈
2. 使用index值遍历表达式
3. 如果发现是一个数字，直接入栈
4. 如果发现符号，分情况：
  4.1 当前符号栈空,直接入栈
  4.2 发现符号栈有操作符,则进行优先级比较，当前操作符优先级小于或者等于栈中操作符，从数栈pop出两个数，在符号栈中pop出一个符号，进行运算，将得到的结果入数栈，然后将当前操作符入符号栈。
  4.3 当前操作符优先级大于栈顶操作符，直接入符号栈
5. 当表达式扫描完毕，就顺序从数栈和符号栈中pop出相应的数和符号，并运算
6. 最后数栈只有一个数字，就是表达式的结果。

```java
package stack;

public class Calculator {
	public static void main(String args[]) {
		// 定义表达式
		String expression = "3+2*6-2";
		ArrayStack2 numStack = new ArrayStack2(10);
		ArrayStack2 operStack = new ArrayStack2(10);
		int index = 0; // 用于扫描
		int num1 = 0;
		int num2 = 0;
		int oper = 0;
		int res = 0;
		char ch = ' ';// 将每次扫描得到的char保存到ch
		
		while(true) {
			ch = expression.substring(index, index + 1).charAt(0);
			if(operStack.isOper(ch)) {// 如果是运算符
				if(!operStack.isEmpty()) {// 符号栈非空，分情况
					// 当前运算符优先级小于或等于栈顶符号优先级
					if(operStack.priority(ch) <= operStack.priority(operStack.peek())) {
						num1 = numStack.pop();
						num2 = numStack.pop();
						oper = operStack.pop();
						res = operStack.cal(num1, num2, oper);
						numStack.push(res);
						operStack.push(ch);
					} else {// 当前操作符优先级大于栈顶符号优先级
						operStack.push(ch);
					}
				} else {
					// 栈空
					operStack.push(ch);
				}
			} else {// 数字
				numStack.push(ch - 48); // '1' => 1 需要减去48；
			}
			// 让index+1并判断是否扫描到expression最后
			index++;
			if(index >= expression.length()) {
				break;
			}
		}
		// 走到这里代表扫描结束
		while(true) {
			// 如果符号栈为空， 数栈只有一个数字 就是结果
			if(operStack.isEmpty()) {
				break; 
			}
			num1 = numStack.pop();
			num2 = numStack.pop();
			oper = operStack.pop();
			res = operStack.cal(num1, num2, oper);
			numStack.push(res);
		}
		System.out.printf("表达式%s = %d", expression, numStack.pop());
	}
}

class ArrayStack2 {
	private int maxSize;
	private int[] stack;
	private int top = -1;

	public ArrayStack2(int maxSize) {
		this.maxSize = maxSize;
		stack = new int[this.maxSize];
	}

	public boolean isFull() {
		return top == maxSize - 1;
	}

	public boolean isEmpty() {
		return top == -1;
	}

	// 入栈
	public void push(int value) {
		if (isFull()) {
			System.out.println("栈满");
			return;
		}
		top++;
		stack[top] = value;
	}

	// 出栈
	public int pop() {
		if (isEmpty()) {
			throw new RuntimeException("栈 空");
		}
		int value = stack[top];
		top--;
		return value;
	}
	
	// 返回当前栈顶的值，但是不出栈
	public int peek() {
		return stack[top];
	}

	// 遍历栈 从栈顶开始显示数据
	public void list() {
		if (isEmpty()) {
			System.out.println("栈空");
			return;
		}
		for (int i = top; i >= 0; i--) {
			System.out.printf("stack[%d]\n", stack[i]);
		}
	}
	
	//返回运算符的优先级
	public int priority(int oper) {
		if(oper == '*' || oper == '-') {
			return 1;
		} else if(oper == '+' || oper == '-') {
			return 0;
		} else {
			return -1;
		}
	}
	
	// 判断是不是一个运算符
	public boolean isOper(char val) {
		return val == '+' || val == '-' || val == '*' || val == '/';
	}
	
	// 运算符
	public int cal(int num1, int num2, int oper) {
		int res = 0;
		switch(oper) {
		case '+':
			res = num1 + num2;
			break;
		case '-':
			res = num2 - num1;
			break;
		case '*':
			res = num1 * num2;
			break;
		case '/':
			res = num2 / num1;
			break;
		default: 
			break;
		}
		return res;
	}
}
```
